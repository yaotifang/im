  /**
   * Autocompletes adresses with default domain names
   *
   * @param {string} user
   * @param {string} provider
   * @return {string}
   */
  autoComplete: function (user, provider) {
    var bits = user.split('@');
    var autodomain = Providers.data[provider].autodomain;
    if (autodomain) {
      user = bits.length > 1 ?
        (bits[0] + '@' + autodomain) :
        (user + '@' + autodomain);
    }
    return user;
  }
